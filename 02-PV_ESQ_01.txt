CREATE TABLESPACE "TBS_DATA_PV" LOGGING
DATAFILE 'C:\oraclexe\app\oracle\oradata\XE\TBS_DATA_PV.dbf' SIZE 1024M
EXTENT MANAGEMENT LOCAL SEGMENT SPACE MANAGEMENT AUTO;


CREATE TABLESPACE "TBS_INDEX_PV" LOGGING
DATAFILE 'C:\oraclexe\app\oracle\oradata\XE\TBS_INDEX_PV.dbf' SIZE 1024M
EXTENT MANAGEMENT LOCAL SEGMENT SPACE MANAGEMENT AUTO;


CREATE USER "GERALDO" PROFILE "DEFAULT" IDENTIFIED BY "admin"
DEFAULT TABLESPACE "TBS_DATA_PV" TEMPORARY TABLESPACE "TEMP" ACCOUNT UNLOCK;
